package table // import "charm.land/bubbles/v2/table"

Package table provides a simple table component for Bubble Tea applications.

TYPES

type Column struct {
	Title string
	Width int
}
    Column defines the table structure.

type KeyMap struct {
	LineUp       key.Binding
	LineDown     key.Binding
	PageUp       key.Binding
	PageDown     key.Binding
	HalfPageUp   key.Binding
	HalfPageDown key.Binding
	GotoTop      key.Binding
	GotoBottom   key.Binding
}
    KeyMap defines keybindings. It satisfies to the help.KeyMap interface,
    which is used to render the help menu.

func DefaultKeyMap() KeyMap
    DefaultKeyMap returns a default set of keybindings.

func (km KeyMap) FullHelp() [][]key.Binding
    FullHelp implements the KeyMap interface.

func (km KeyMap) ShortHelp() []key.Binding
    ShortHelp implements the KeyMap interface.

type Model struct {
	KeyMap KeyMap
	Help   help.Model

	// Has unexported fields.
}
    Model defines a state for the table widget.

func New(opts ...Option) Model
    New creates a new model for the table widget.

func (m *Model) Blur()
    Blur blurs the table, preventing selection or movement.

func (m Model) Columns() []Column
    Columns returns the current columns.

func (m Model) Cursor() int
    Cursor returns the index of the selected row.

func (m *Model) Focus()
    Focus focuses the table, allowing the user to move around the rows and
    interact.

func (m Model) Focused() bool
    Focused returns the focus state of the table.

func (m *Model) FromValues(value, separator string)
    FromValues create the table rows from a simple string. It uses `\n` by
    default for getting all the rows and the given separator for the fields on
    each row.

func (m *Model) GotoBottom()
    GotoBottom moves the selection to the last row.

func (m *Model) GotoTop()
    GotoTop moves the selection to the first row.

func (m Model) Height() int
    Height returns the viewport height of the table.

func (m Model) HelpView() string
    HelpView is a helper method for rendering the help menu from the keymap.
    Note that this view is not rendered by default and you must call it manually
    in your application, where applicable.

func (m *Model) MoveDown(n int)
    MoveDown moves the selection down by any number of rows. It can not go below
    the last row.

func (m *Model) MoveUp(n int)
    MoveUp moves the selection up by any number of rows. It can not go above the
    first row.

func (m Model) Rows() []Row
    Rows returns the current rows.

func (m Model) SelectedRow() Row
    SelectedRow returns the selected row. You can cast it to your own
    implementation.

func (m *Model) SetColumns(c []Column)
    SetColumns sets a new columns state.

func (m *Model) SetCursor(n int)
    SetCursor sets the cursor position in the table.

func (m *Model) SetHeight(h int)
    SetHeight sets the height of the viewport of the table.

func (m *Model) SetRows(r []Row)
    SetRows sets a new rows state.

func (m *Model) SetStyles(s Styles)
    SetStyles sets the table styles.

func (m *Model) SetWidth(w int)
    SetWidth sets the width of the viewport of the table.

func (m Model) Update(msg tea.Msg) (Model, tea.Cmd)
    Update is the Bubble Tea update loop.

func (m *Model) UpdateViewport()
    UpdateViewport updates the list content based on the previously defined
    columns and rows.

func (m Model) View() string
    View renders the component.

func (m Model) Width() int
    Width returns the viewport width of the table.

type Option func(*Model)
    Option is used to set options in New. For example:

        table := New(WithColumns([]Column{{Title: "ID", Width: 10}}))

func WithColumns(cols []Column) Option
    WithColumns sets the table columns (headers).

func WithFocused(f bool) Option
    WithFocused sets the focus state of the table.

func WithHeight(h int) Option
    WithHeight sets the height of the table.

func WithKeyMap(km KeyMap) Option
    WithKeyMap sets the key map.

func WithRows(rows []Row) Option
    WithRows sets the table rows (data).

func WithStyles(s Styles) Option
    WithStyles sets the table styles.

func WithWidth(w int) Option
    WithWidth sets the width of the table.

type Row []string
    Row represents one line in the table.

type Styles struct {
	Header   lipgloss.Style
	Cell     lipgloss.Style
	Selected lipgloss.Style
}
    Styles contains style definitions for this list component. By default,
    these values are generated by DefaultStyles.

func DefaultStyles() Styles
    DefaultStyles returns a set of default style definitions for this table.
